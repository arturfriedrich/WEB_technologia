void nevek(const char* regi, char* uj, 
           const char* mit, const char* mire) {
  const char *eleje = regi;
  char *vege;
  int mitHossz = strlen(mit);
  while((vege = strstr(eleje, mit)) != NULL) {
    strncat(uj, eleje, vege-eleje);
    strcat(uj, mire);
    eleje = vege + mitHossz;
  }
  strcat(uj, eleje);
}
